  SUPABASE_USER_PASSWORD=password

  Option C: Custom API Key
  // Create dedicated service account

  Option A: Mirror Supabase SDK
  // Familiar to Supabase users

  // More powerful, direct SQL
  zephix_db_query({

    params: [true]

  // Both options available
